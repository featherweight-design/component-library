# Input

---

<Story id="components-form-input--default" />

---

A basic `Input` component with `disabled` and `error` states, as well as `min/max` values and an accessible `label`.

## Quick Start ⚡️

```jsx
import { Input } from '@f-design/component-library';

const MyComponent = () => {
  // ...things
  const [value, handleChangeTextValue] = useState('');

  const mockOnChange = ({
    target: { value },
  }: ChangeEvent<HTMLInputElement>): void => handleChangeTextValue(value);

  return (
    <Input value={value} name="default" label="Label" onChange={mockOnChange} />
  );
};
```
